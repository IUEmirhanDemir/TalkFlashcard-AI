108    def on_space_pressed(self, event=None):
109        """
110        This method is triggered when the space key is pressed.
111
112        Logic:
113        - If TTS (is_speaking) or STT (is_transcribing) is active, ignore the space press.
114        - If we are not currently recording, ignore the space press.
115        - Otherwise, stop the recording to accept the user's answer early.
116        """
117        if self.is_speaking or self.is_transcribing:
118            print("Leertaste ignoriert: KI spricht oder transkribiert noch.")
119            return
120
121        if not self.recording:
122            print("Leertaste ignoriert: Keine Aufnahme aktiv.")
123            return
124
125        # If we're here, the user is speaking (is_listening=True), and we can stop early
126        self.stop_recording()

231    def speak_text(self, text, callback=None, listen_after_speech=True):
232        """
233        Converts the given text into speech (TTS) using OpenAI (model='tts-1')
234        and plays it back via a platform-specific audio player.
235
236        :param text: The text to be spoken in German.
237        :param callback: A callable executed after speech playback finishes.
238        :param listen_after_speech: If True, listening for a user answer starts automatically.
239        """
240        if self.is_speaking or self.window_closed:
241            return
242
243        def run_tts():
244            if self.window_closed:
245                return
246            try:
247                self.is_speaking = True
248                response = openai.audio.speech.create(
249                    model="tts-1",
250                    voice="nova",
251                    input=text,
252                    speed=1.1
253                )
254                audio_data = response.content
255
256                with open(self.speech_file_path, "wb") as speech_file:
257                    speech_file.write(audio_data)
258
259                print(f"Audio gespeichert unter: {self.speech_file_path}")
260
261                if not self.window_closed:
262                    # If something is still playing, terminate it
263                    if self.playback_process and self.playback_process.poll() is None:
264                        self.playback_process.terminate()
265
266                    # Platform-specific audio playback
267                    if sys.platform == 'win32':
268                        self.playback_process = subprocess.Popen(
269                            ['cmd', '/c', f'start /min wmplayer "{self.speech_file_path}"'],
270                            stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL
271                        )
272                    elif sys.platform == 'darwin':
273                        self.playback_process = subprocess.Popen(
274                            ['afplay', self.speech_file_path],
275                            stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL
276                        )
277                    else:
278                        self.playback_process = subprocess.Popen(
279                            ['aplay', self.speech_file_path],
280                            stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL
281                        )
282
283                    self.playback_process.wait()
284
285            except Exception as e:
286                self.safe_display_message(f"Fehler bei der Sprachausgabe: {str(e)}", "System")
287                print(f"Fehler bei der Sprachausgabe: {str(e)}")
288            finally:
289                self.is_speaking = False
290                self.playback_process = None
291                if callback and not self.window_closed:
292                    self.after(0, callback)
293                elif listen_after_speech and not self.window_closed:
294                    self.listen_for_answer()
295
296        threading.Thread(target=run_tts, daemon=True).start()

298    def listen_for_answer(self):
299        """
300        Starts listening for the user's spoken answer.
301        Records up to 60 seconds of audio at 48 kHz, then transcribes the result with Whisper.
302
303        We do NOT set is_transcribing to True until AFTER the recording ends,
304        so the user can press space to stop early.
305        """
306        if self.is_listening or self.window_closed:
307            return
308
309        def run_stt():
310            if self.window_closed:
311                return
312            try:
313                # Start listening/recording, so the user can press space
314                self.is_listening = True
315                self.safe_display_message(
316                    "Bitte antworte jetzt mündlich (max. 60 Sekunden). Drücke die Leertaste zum Beenden.",
317                    "KI-Lernpartner"
318                )
319                print("Start der Audioaufnahme...")
320
321                duration = 60
322                fs = 48000  # Higher sample rate for better audio quality
323                channels = 1
324
325                # Record audio for up to 60 seconds or until stopped by space
326                recording = sd.rec(int(duration * fs), samplerate=fs, channels=channels, dtype='float32')
327                self.recording = True
328
329                # Keep waiting until user stops (self.recording=False) or window closes
330                while self.recording and not self.window_closed:
331                    sd.sleep(100)
332                sd.stop()
333
334                # Recording phase is done
335                self.is_listening = False
336
337                if self.window_closed:
338                    print("Fenster bereits geschlossen, breche ab.")
339                    return
340
341                sf.write(self.recorded_audio_path, recording, fs)
342                print(f"Audio gespeichert unter: {self.recorded_audio_path}")
343
344                if not os.path.exists(self.recorded_audio_path) or os.path.getsize(self.recorded_audio_path) == 0:
345                    raise ValueError("Die aufgezeichnete Audiodatei existiert nicht oder ist leer.")
346
347                # Now we set is_transcribing = True to block space if user tries again
348                self.is_transcribing = True
349
350                # Validate the audio file
351                with wave.open(self.recorded_audio_path, 'rb') as wf:
352                    assert wf.getnchannels() == 1, "Audio ist nicht mono."
353                    assert wf.getframerate() == 48000, "Audio hat nicht die richtige Abtastrate (erwartet: 48000)."
354                    assert wf.getsampwidth() == 2, "Audio hat nicht die richtige Bit-Tiefe."
355                print("Audio-Datei validiert.")
356
357                # Transcribe with Whisper
358                with open(self.recorded_audio_path, "rb") as audio_file:
359                    transcription = openai.audio.transcriptions.create(
360                        model="whisper-1",
361                        file=audio_file
362                    )
363                user_response = transcription.text.strip()
364                self.safe_display_message(f"Transkription: {user_response}", "Nutzer")
365
366                if not user_response:
367                    self.safe_display_message(
368                        "Keine Antwort erkannt. Versuchen wir es mit der nächsten Frage",
369                        "KI-Lernpartner"
370                    )
371                    self.ask_next_question()
372                    return
373
374                self.evaluate_response(user_response)
375
376            except AssertionError as ae:
377                self.safe_display_message(f"Audio Validierungsfehler: {str(ae)}", "System")
378                print(f"Audio Validierungsfehler: {str(ae)}")
379            except Exception as e:
380                self.safe_display_message(f"Fehler bei der Aufnahme oder Verarbeitung: {str(e)}", "System")
381                print(f"Fehler bei der Aufnahme oder Verarbeitung: {str(e)}")
382            finally:
383                self.is_transcribing = False
384
385        threading.Thread(target=run_stt, daemon=True).start()

299    def listen_for_answer(self):

301        Starts listening for the user's spoken answer.
302        Records up to 60 seconds of audio at 48 kHz, then transcribes the result with Whisper.
303

307        if self.is_listening or self.window_closed:
308            return
309
310        def run_stt():
311            if self.window_closed:
312                return

322                duration = 60

324                channels = 1

326                # Record audio for up to 60 seconds or until stopped by space

331                while self.recording and not self.window_closed:
