 41    def truncate_text(self, text, max_length):
 42        """
 43        Truncates the given text to the specified length, ensuring no line breaks.
 44
 45        Args:
 46            text (str): The text to be truncated.
 47            max_length (int): The maximum length of the text.
 48
 49        Returns:
 50            str: The truncated text.
 51        """
 52        text = text.replace('\n', ' ').replace('\r', ' ')
 53        text = ' '.join(text.split())
 54        if len(text) > max_length:
 55            return text[:max_length].rstrip() + "..."
 56        return text

282    def on_right_click(self, event):
283        """
284        Displays the right-click context menu for deleting selected flashcards.
285        """
286        selected_items = self.flashcards_tree.selection()
287        if selected_items:

289            context_menu.add_command(

291            context_menu.tk_popup(event.x_root, event.y_root)
292        else:
293            pass

295    def delete_selected_flashcards(self):
296        """
297        Deletes selected flashcards after confirmation.
298        """
299        selected_items = self.flashcards_tree.selection()
300        if selected_items:
301            confirm = messagebox.askyesno(
302                "Karteikarten löschen",
303                f"Möchten Sie die ausgewählten Karteikarten wirklich löschen?"
304            )
305            if confirm:
306                for item_id in selected_items:
307                    try:
308                        flashcard_id = int(item_id)
309                        self.db_service.delete_flashcard(flashcard_id)
310                        self.flashcards_tree.delete(item_id)
311                    except Exception as e:
312                        messagebox.showerror(
313                            "Fehler", f"Fehler beim Löschen der Karteikarte: {e}")
314                messagebox.showinfo(
315                    "Erfolg", "Die ausgewählten Karteikarten wurden gelöscht.")
316        else:
317            messagebox.showwarning(
318                "Warnung", "Keine Karteikarten ausgewählt.")
319
320    def on_double_click(self, event):
321        """
322        Edits the selected flashcard on a double-click event.
323        """
324        selected_item = self.flashcards_tree.selection()
325        if selected_item:
326            selected_item_id = selected_item[0]
327
328            if selected_item_id == "no_flashcards":
329                return
330
331            try:
332                flashcard_id = int(selected_item_id)
333                flashcard = next(
334                    (fc for fc in self.db_service.get_flashcards_by_module(
335                        self.module.id) if fc.id == flashcard_id),
336                    None
337                )
338                if flashcard:
339                    self.module_controller.edit_flashcard(flashcard)
340            except ValueError:
341                return

362        Starts the normal learning mode if flashcards are available.
363        """

366                messagebox.showinfo(

370            normal_mode_controller = NormalModeController(

def truncate_text(self, text, max_length): 
42    def truncate_text(self, text, max_length):
43        """
44        Truncates the given text to the specified length, ensuring no line breaks.
45
46        Args:
47            text (str): The text to be truncated.
48            max_length (int): The maximum length of the text.
49
50        Returns:
51            str: The truncated text.
52        """
53        text = text.replace('\n', ' ').replace('\r', ' ')
54        text = ' '.join(text.split())
55        if len(text) > max_length:
56            return text[:max_length].rstrip() + "..."
57        return text
